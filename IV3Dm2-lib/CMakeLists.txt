cmake_minimum_required(VERSION 3.0.0)
project(IV3Dm-lib)

INCLUDE_DIRECTORIES(${CMAKE_SOURCE_DIR})
INCLUDE_DIRECTORIES(${CMAKE_SOURCE_DIR}/mocca/mocca/include)
INCLUDE_DIRECTORIES(${CMAKE_SOURCE_DIR}/mocca/mocca/src/thirdparty)

FILE(GLOB COMMON_HEADER ${PROJECT_SOURCE_DIR}/Common/*.h)
FILE(GLOB COMMON_SOURCE ${PROJECT_SOURCE_DIR}/Common/*.cpp)
FILE(GLOB COMMUNICATION_HEADER ${PROJECT_SOURCE_DIR}//*.h)
FILE(GLOB COMMUNICATION_SOURCE ${PROJECT_SOURCE_DIR}/Communication/*.cpp)
FILE(GLOB SCENE_HEADER ${PROJECT_SOURCE_DIR}/Scene/*.h)
FILE(GLOB SCENE_SOURCE ${PROJECT_SOURCE_DIR}/Scene/*.cpp)

ADD_LIBRARY(IV3Dm-lib 
	${COMMON_HEADER} ${COMMON_SOURCE}
	${COMMUNICATION_HEADER} ${COMMUNICATION_SOURCE}
	${SCENE_HEADER} ${SCENE_SOURCE}
)
